$('input[name="swift_user_type"]').on('change', function() {
        var selectedOption = $('input[name="swift_user_type"]:checked').val();
        if(selectedOption === "individual"){
            $('#first_name').attr('required',true);
            $('#last_name').attr('required',true);
            $('#dynamic_label1').removeAttr('required');
            $(".individual").show();
            $(".show_name").hide();
        }else if(selectedOption === "company"){
            $("#dynamic_label").html('Business Name');
            $('#dynamic_label1').prop('placeholder', "Business Name");

            $('#first_name').removeAttr('required');
            $('#last_name').removeAttr('required');

            $('#dynamic_label1').attr('required',true);


            $(".show_name").show();
            $(".individual").hide();

        }
        else if(selectedOption === "joint" || selectedOption === "corporatetrust" || selectedOption === "individualtrust"){
            $("#dynamic_label").html('Name');
            $('#dynamic_label1').prop('placeholder', "Name");

            $('#first_name').removeAttr('required');
            $('#last_name').removeAttr('required');

            $('#dynamic_label1').attr('required',true);

            $(".show_name").show();
            $(".individual").hide();
        }
        else{
            $('#first_name').attr('required',true);
            $('#last_name').attr('required',true);

            $('#dynamic_label1').removeAttr('required');

            $(".individual").show();
            $(".show_name").hide();
        }
    });
